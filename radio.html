document.addEventListener("DOMContentLoaded", function() { const audio = document.getElementById("audio"); const prevButton = document.getElementById("prev"); const nextButton = document.getElementById("next"); const playlist = [ "Sna Taylor Swift.mp3", "bruno.mp3", "j e m.mp3", "james.mp3", "jem.mp3", "jm.mp3", "jorge e mateus.mp3", "juju.mp3", "justin.mp3", "magico.mp3" ]; let currentSongIndex = 0; function playSong(index) { if (index >= 0 && index < playlist.length) { audio.src = playlist[index]; audio.play(); currentSongIndex = index; } } playSong(currentSongIndex); prevButton.addEventListener("click", function() { currentSongIndex = (currentSongIndex - 1 + playlist.length) % playlist.length; playSong(currentSongIndex); }); nextButton.addEventListener("click", function() { currentSongIndex = (currentSongIndex + 1) % playlist.length; playSong(currentSongIndex); }); });
